## Pester

```inventory
OsBasic:
  hostname: "test"
  installLang: "0411"
```

```pester.ps1
import-module pester
import-module powershell-yaml

$yamlContent = Get-content -raw -path ./inventory.yml -Encoding utf8
$config = ConvertFrom-Yaml $yamlContent

$osBasicCases = @(
  @{
    key = 'hostname'
    expected = $config.OsBasic.hostname
    actualScript = { $env:COMPUTERNAME }
  }
  @{
    key = 'installLang'
    expected = $config.OsBasic.installLang
    actualScript = { ('{0:D4}' -f (Get-WinSystemLocale).LCID) }
  }
)

Describe "sample-test" {
  It "<key> should match (expected: <expected>)" -ForEach $osBasicCases {
    param($key, $expected, $actualScript)

    $actual = & $actualScript

    Write-Verbose "key=[$key] expected=[$expected] actual=[$actual]"
    $actual | Should -Be $expected
  }
}
```

```powershell
Invoke-Pester -Path .\sample.tests.ps1 -Verbose -Output Detailed
```
