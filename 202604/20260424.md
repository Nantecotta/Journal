```pester.ps1
import-module pester
import-module powershell-yaml

$yamlContent = Get-content -raw -path ./inventory.yml -Encoding utf8
$config = ConvertFrom-Yaml $yamlContent

function Find-InstalledSoftware {
  param(
    [Parameter(Mandatory = $true)]
    [string]$Name
  )

  $uninstallPaths = @(
    'HKLM:\SOFTWARE\Microsoft\Windows\CurrentVersion\Uninstall\*',
    'HKLM:\SOFTWARE\WOW6432Node\Microsoft\Windows\CurrentVersion\Uninstall\*',
    'HKCU:\SOFTWARE\Microsoft\Windows\CurrentVersion\Uninstall\*',
    'HKCU:\SOFTWARE\WOW6432Node\Microsoft\Windows\CurrentVersion\Uninstall\*'
  )

  foreach ($path in $uninstallPaths) {
    Get-ItemProperty -Path $path -ErrorAction SilentlyContinue |
      Where-Object { $_.DisplayName -and $_.DisplayName -like "*$Name*" }
  }
}

$osBasicCases = @(
  @{
    key = 'hostname'
    expected = $config.OsBasic.hostname
    actualScript = { $env:COMPUTERNAME }
  }
  @{
    key = 'installLang'
    expected = $config.OsBasic.installLang
    actualScript = { ('{0:D4}' -f (Get-WinSystemLocale).LCID) }
  }
)

$winPatchCases = @(
  foreach ($patch in $config.OsBasic.winPatches) {
    @{
      name = $patch.name
      state = [bool]$patch.state
    }
  }
)

$diskCases = @(
  foreach ($disk in $config.Disks) {
    $expectedSizeGB = switch ($disk.diskSizeUnit) {
      'GB' { [double]$disk.diskSize }
      'TB' { [double]$disk.diskSize * 1024 }
      default { throw "Unknown diskSizeUnit: $($disk.diskSizeUnit)" }
    }

    @{
      diskpurpose = $disk.diskpurpose
      diskNumber = [int]$disk.diskNumber
      driveLetter = [string]$disk.driveLetter
      expectedSizeGB = [math]::Round($expectedSizeGB, 1)
    }
  }
)

$hostCases = @(
  if ($null -ne $config.hosts) {
    foreach ($entry in $config.hosts) {
      @{ entry = [string]$entry }
    }
  }
)

$softwareCases = @(
  foreach ($sw in $config.softwares) {
    @{
      name = [string]$sw.name
      state = [bool]$sw.state
    }
  }
)

Describe "sample-test" {
  It "<key> should match (expected: <expected>)" -ForEach $osBasicCases {
    param($key, $expected, $actualScript)

    $actual = & $actualScript

    Write-Verbose "key=[$key] expected=[$expected] actual=[$actual]"
    $actual | Should -Be $expected
  }
}

Describe "winPatches" {
  It "patch <name> installed should be <state>" -ForEach $winPatchCases {
    param($name, $state)

    $actual = [bool](Get-HotFix -Id $name -ErrorAction SilentlyContinue | Select-Object -First 1)

    Write-Verbose "name=[$name] expected=[$state] actual=[$actual]"
    $actual | Should -Be $state
  }
}

Describe "disks" {
  It "disk[<diskNumber>] <diskpurpose>: driveLetter should be <driveLetter>" -ForEach $diskCases {
    param($diskNumber, $diskpurpose, $driveLetter, $expectedSizeGB)

    $partition = Get-Partition -DiskNumber $diskNumber -ErrorAction Stop |
      Where-Object { $_.DriveLetter -eq $driveLetter }
    $actual = $partition.DriveLetter

    Write-Verbose "diskNumber=[$diskNumber] expectedDriveLetter=[$driveLetter] actual=[$actual]"
    $actual | Should -Be $driveLetter
  }

  It "disk[<diskNumber>] <diskpurpose>: size should be <expectedSizeGB>GB" -ForEach $diskCases {
    param($diskNumber, $diskpurpose, $driveLetter, $expectedSizeGB)

    $disk = Get-Disk -Number $diskNumber -ErrorAction Stop
    $actualSizeGB = [math]::Round($disk.Size / 1GB, 1)

    Write-Verbose "diskNumber=[$diskNumber] expectedSizeGB=[$expectedSizeGB] actualSizeGB=[$actualSizeGB]"
    $actualSizeGB | Should -Be $expectedSizeGB
  }
}

Describe "hosts" {
  $hostsFilePath = "$env:SystemRoot\System32\drivers\etc\hosts"

  It "hosts entries are optional and current inventory has none" -Skip:($hostCases.Count -ne 0) {
    $hostCases.Count | Should -Be 0
  }

  It "hosts should contain <entry>" -ForEach $hostCases {
    param($entry)

    $pattern = '^\s*' + ([regex]::Escape($entry) -replace '\\ ', '\s+') + '\s*$'
    $actual = Select-String -Path $hostsFilePath -Pattern $pattern -Quiet

    Write-Verbose "entry=[$entry] actual=[$actual]"
    $actual | Should -BeTrue
  }
}

Describe "softwares" {
  It "<name> installed should be <state>" -ForEach $softwareCases {
    param($name, $state)

    $actual = [bool](Find-InstalledSoftware -Name $name | Select-Object -First 1)

    Write-Verbose "name=[$name] expected=[$state] actual=[$actual]"
    $actual | Should -Be $state
  }
}
```
